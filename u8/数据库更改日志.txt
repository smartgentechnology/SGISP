---lc_pprocedure(生产工序表)
(flag:0表示未完成，1表示已完成 ，2表示焊接入库，3表示初试入库，4表示出厂入库，5表示报废，6表示包装入库，7表示焊接报废入库，8表示初试报废入库，9表示出厂报废入库,10 表示待包装入库，11表示包装完入库)
(type:1表示焊接工序，2表示初试工序 ，3表示出厂工序，4表示包装工序，5表示成品库，6表示在制品库，7表示电控库，8表示待料库)
create table lc_pprocedure(
id int(10) unsigned not null primary key auto_increment,
model varchar(100) not null default '',
batch varchar(100) not null default '',
starttime int(10) unsigned not null default 0,
endtime int(10) unsigned not null default 0,
number int(10) unsigned not null default 0,
hours float(5,1) unsigned not null default 0,
people int(10) unsigned not null default 1,
sn varchar(100) not null default '',
manager_id int(10) not null default 0,
type int(10) unsigned not null default 0,
flag int(4) unsigned not null default 0,
remark varchar(100) not null default ''
)engine myisam charset utf8;
